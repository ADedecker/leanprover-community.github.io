import tactic

def equivalency (p q : Prop) : Prop := (p → q) ∧ (q → p)

open_locale classical

theorem x (p q: Prop) : equivalency p q → ((p ∧ q) ∨ ((¬ p) ∧ (¬ q))) :=
begin
  intro eq,
  unfold equivalency at eq,
  cases eq,
  by_cases hp : p; by_cases hq : q,
    finish,
    finish,
    finish,
    finish,
end
