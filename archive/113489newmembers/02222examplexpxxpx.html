example : ¬ (∀ x, ¬ p x) → (∃ x, p x) :=
    (assume hnAxnpx : ¬ (∀ x, ¬ p x),
        by_contradiction
            (λ hnExpx : ¬ (∃ x, p x),

Johan Commelin (Nov 04 2019 at 18:19):

@Eduardo Cavazos Here's a tactic mode proof that doesn't just kill it with automation.

import tactic
universe variable u
open_locale classical
variables {X : Type u} (p : X → Prop)

example : ¬ (∀ x, ¬ p x) → (∃ x, p x) :=
begin
  contrapose!,
  exact id,
end
