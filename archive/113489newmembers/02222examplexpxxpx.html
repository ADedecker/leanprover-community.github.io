example : ¬ (∀ x, ¬ p x) → (∃ x, p x) :=
    (assume hnAxnpx : ¬ (∀ x, ¬ p x),
        by_contradiction
            (λ hnExpx : ¬ (∃ x, p x),
                (hnAxnpx
                    (λ x, (λ hpx,
                        hnExpx (exists.intro x hpx))))))
