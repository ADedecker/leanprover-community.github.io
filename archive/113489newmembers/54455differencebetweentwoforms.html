example (h : p ∨ q) : q ∨ p :=

  (assume hq : q,
    show q ∨ p, from or.intro_left p hq)

and this one that doesn't?

example : p ∨ q → q ∨ p :=
or.elim
  (assume hp : p,
    show q ∨ p, from or.intro_right q hp)
  (assume hq : q,
    show q ∨ p, from or.intro_left p hq)
