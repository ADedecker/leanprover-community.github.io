lemma finset.image_ne_empty {V W : Type} [decidable_eq W] {f : V -> W} {A : finset V}
        (h : A ≠ ∅) : finset.image f A ≠ ∅
    := Exists.dcases_on (finset.exists_mem_of_ne_empty h)
        (λ _, finset.ne_empty_of_mem ∘ finset.mem_image_of_mem f)
