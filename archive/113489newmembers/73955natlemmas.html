Reid Barton (May 12 2019 at 23:40):

lemma sub_add_eq_add_sub_of_le {a b c : ℕ} (h : b ≤ a) : (a - b) + c = (a + c) - b :=
begin
  symmetry,
  apply nat.sub_eq_of_eq_add,
  rw [←add_assoc, nat.add_sub_cancel' h],
end
