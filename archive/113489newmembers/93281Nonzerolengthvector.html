This works for me:

namespace hidden

inductive vector (α : Type) : ℕ → Type
| nil {} : vector 0
| cons {n} : α → vector n → vector (n+1)

def prod {F G : Type}(f : F → G → G) (g : G → G → G):
       Π {n : ℕ},
       vector G (n + 1) →
       vector F (n + 1) → G
      | (0) (vector.cons h hs) (vector.cons m ms) :=  f m h
      | (n + 1)(vector.cons h hs) (vector.cons m ms) :=
           g (f m h) (prod hs ms)

end hidden
