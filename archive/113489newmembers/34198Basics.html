Why does this fail, and how to fix it?

def show_p : (p ∧ q) ∨ (p ∧ r) → p :=
  (assume h: (p ∧ q) ∨ (p ∧ r),
  show p,
  from or.elim h
  and.left
  and.left
)

def show_q_or_r : (p ∧ q) ∨ (p ∧ r) → q ∨ r :=
(assume h: (p ∧ q) ∨ (p ∧ r),
  show q ∨ r,
  from or.elim h
  (λ z, or.inl  (and.right z))
  (λ z, or.inr (and.right z))
)

example : (p ∧ q) ∨ (p ∧ r) → p ∧ (q ∨ r) :=
(assume h: (p ∧ q) ∨ (p ∧ r) ,
show p ∧ (q ∨ r),
from and.intro ((show_p h ) ∧ (show_q_or_r h))
)
