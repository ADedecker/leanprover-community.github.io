  coe := λ f x, eval x f}

def 𝕍 (S : set (mv_polynomial n k)) : set (n → k) :=
-- {x : n → k | ∀ s ∈ S, s x = 0} -- doesn't work

{x : n → k | ∀ s ∈ S, eval x s = 0} -- works BUT VARIABLES IN STUPID ORDER!
